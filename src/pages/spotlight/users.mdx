export function MaybeImg({ src, alt }) {
  if (src) {
    return <img src={src} alt={alt} itemProp="image" />;
  }
  return <span className={clsx(styles.bigOrg)}>{alt}</span>;
}

export function User({ alt, href, asset }) {
  return (
    <div className={clsx("col", styles.userLogo)}>
      <Link to={href}>
        <MaybeImg src={asset} alt={alt} />
      </Link>
    </div>
  );
}
